// This sample config produces three clones of the stokc Wheesley engine, each with the following plumes attached.

// Subsonic Hypno
+PART[JetEngine]
{
	@name = JetEngine2
	@title = J-33B "Wheesley Hypno" Turbofan
	@EFFECTS
	{
		@running_thrust
		{
			!PREFAB_PARTICLE {}
			MODEL_MULTI_PARTICLE
			{
				name = fire
				modelName = PlumeParty/Engines/Turbine/subHypno
				transformName = thrustTransform
				emission = 0.0 0.0
				emission = 0.6 1.0
				speed = 0.0 0.8
				speed = 1.0 1.0
				localPosition = 0, 0, -0.5
			}
			MODEL_MULTI_PARTICLE
			{
				name = smoke
				modelName = PlumeParty/Engines/Turbine/petroTrail
				transformName = thrustTransform
				emission = 0.1 0.0
				emission = 0.5 1.0
				speed = 0.0 0.8
				speed = 1.0 1.0
				localPosition = 0, 0, -0.5
			}
		}
	}
}
